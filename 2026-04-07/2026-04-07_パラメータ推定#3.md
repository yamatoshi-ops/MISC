以下は、**PSなし / PSありを同一条件で比較**できる `simulation.py` です。
`plant.py` と `identifier_d.py` はそのまま使えます。

この版では、

* ケースA: **PSなし**
* ケースB: **PSあり**（d軸に正弦波重畳）

を順に実行して、推定値・電流応答・(\zeta) 信号を比較します。
文献でいう「PS性がないと複数パラメータ同定が難しい」という点を、まず数値で見るための版です。 

---

## `simulation.py`

```python
# simulation.py
from __future__ import annotations

import math
from dataclasses import dataclass

import matplotlib.pyplot as plt
import numpy as np

from plant import PMSMParams, PMSMPlantDQ
from identifier_d import IdentifierConfig, ThreeParameterIdentifierD


@dataclass
class SimCaseConfig:
    name: str
    enable_ps: bool
    Ts: float = 1.0e-4
    T_end: float = 1.5
    omega: float = 400.0          # electrical angular speed [rad/s]
    id_base: float = -40.0        # [A]
    iq_ref: float = 80.0          # [A]
    omega_ps: float = 50.0        # [rad/s]
    id_ps_amp: float = 10.0       # [A]
    v_limit: float = 300.0        # [V]


def run_case(cfg: SimCaseConfig) -> dict:
    Ts = cfg.Ts
    steps = int(cfg.T_end / Ts)
    t = np.arange(steps) * Ts

    # True parameters
    true_params = PMSMParams(
        R1=0.03,
        Ld=1.00e-3,
        Lq=1.50e-3,
        Phi=0.10,
    )

    plant = PMSMPlantDQ(true_params, Ts)

    ident = ThreeParameterIdentifierD(
        IdentifierConfig(
            Ts=Ts,
            omega_ss3=2.0 * cfg.omega_ps,
            lambda1=0.99,
            lambda2=1.0,
            gsig=100.0,
            nominal_R1=0.03,
            nominal_Ld=1.00e-3,
            nominal_Lq=1.50e-3,
            use_output_lpf=False,
            omega_out=100.0,
        )
    )

    # Minimal PI current controller
    kp_d = 8.0
    ki_d = 400.0
    kp_q = 8.0
    ki_q = 400.0

    int_d = 0.0
    int_q = 0.0

    # Logs
    id_log = np.zeros(steps)
    iq_log = np.zeros(steps)
    id_ref_log = np.zeros(steps)
    iq_ref_log = np.zeros(steps)
    id_ps_log = np.zeros(steps)

    vd_log = np.zeros(steps)
    vq_log = np.zeros(steps)

    R1_hat_log = np.zeros(steps)
    Ld_hat_log = np.zeros(steps)
    Lq_hat_log = np.zeros(steps)
    err_log = np.zeros(steps)

    zeta0_log = np.zeros(steps)
    zeta1_log = np.zeros(steps)
    zeta2_log = np.zeros(steps)
    zeta3_log = np.zeros(steps)

    gamma11_log = np.zeros(steps)
    gamma22_log = np.zeros(steps)
    gamma33_log = np.zeros(steps)

    torque_log = np.zeros(steps)

    for k in range(steps):
        tk = t[k]

        if cfg.enable_ps:
            id_ps = cfg.id_ps_amp * math.sin(cfg.omega_ps * tk)
        else:
            id_ps = 0.0

        id_ref = cfg.id_base + id_ps
        iq_ref = cfg.iq_ref

        e_d = id_ref - plant.id
        e_q = iq_ref - plant.iq

        int_d += Ts * e_d
        int_q += Ts * e_q

        # Feedforward/decoupling using true model
        vd_ff = true_params.R1 * plant.id - cfg.omega * true_params.Lq * plant.iq
        vq_ff = (
            true_params.R1 * plant.iq
            + cfg.omega * true_params.Ld * plant.id
            + cfg.omega * true_params.Phi
        )

        vd = vd_ff + kp_d * e_d + ki_d * int_d
        vq = vq_ff + kp_q * e_q + ki_q * int_q

        vd = float(np.clip(vd, -cfg.v_limit, cfg.v_limit))
        vq = float(np.clip(vq, -cfg.v_limit, cfg.v_limit))

        id_now, iq_now = plant.step(vd, vq, cfg.omega)
        torque = plant.electromagnetic_torque(pole_pairs=1)

        est = ident.update(vd=vd, id_=id_now, iq=iq_now, omega=cfg.omega)

        id_log[k] = id_now
        iq_log[k] = iq_now
        id_ref_log[k] = id_ref
        iq_ref_log[k] = iq_ref
        id_ps_log[k] = id_ps

        vd_log[k] = vd
        vq_log[k] = vq

        R1_hat_log[k] = est["R1_hat"]
        Ld_hat_log[k] = est["Ld_hat"]
        Lq_hat_log[k] = est["Lq_hat"]
        err_log[k] = est["error"]

        zeta0_log[k] = est["zeta0"]
        zeta1_log[k] = est["zeta1"]
        zeta2_log[k] = est["zeta2"]
        zeta3_log[k] = est["zeta3"]

        gamma11_log[k] = est["Gamma_diag"][0]
        gamma22_log[k] = est["Gamma_diag"][1]
        gamma33_log[k] = est["Gamma_diag"][2]

        torque_log[k] = torque

    return {
        "cfg": cfg,
        "t": t,
        "true_params": true_params,
        "id": id_log,
        "iq": iq_log,
        "id_ref": id_ref_log,
        "iq_ref": iq_ref_log,
        "id_ps": id_ps_log,
        "vd": vd_log,
        "vq": vq_log,
        "R1_hat": R1_hat_log,
        "Ld_hat": Ld_hat_log,
        "Lq_hat": Lq_hat_log,
        "error": err_log,
        "zeta0": zeta0_log,
        "zeta1": zeta1_log,
        "zeta2": zeta2_log,
        "zeta3": zeta3_log,
        "Gamma11": gamma11_log,
        "Gamma22": gamma22_log,
        "Gamma33": gamma33_log,
        "torque": torque_log,
    }


def summarize_result(result: dict) -> None:
    p = result["true_params"]
    print(f"=== {result['cfg'].name} ===")
    print(f"R1 true={p.R1:.6f}, hat={result['R1_hat'][-1]:.6f}")
    print(f"Ld true={p.Ld:.6e}, hat={result['Ld_hat'][-1]:.6e}")
    print(f"Lq true={p.Lq:.6e}, hat={result['Lq_hat'][-1]:.6e}")
    print(f"final error={result['error'][-1]:.6e}")
    print()


def plot_comparison(res_no_ps: dict, res_ps: dict) -> None:
    t = res_no_ps["t"]
    p = res_no_ps["true_params"]

    fig, axes = plt.subplots(5, 2, figsize=(14, 18), sharex="col")

    # Row 1: current response
    axes[0, 0].plot(t, res_no_ps["id_ref"], label="id_ref")
    axes[0, 0].plot(t, res_no_ps["id"], label="id")
    axes[0, 0].plot(t, res_no_ps["iq_ref"], label="iq_ref")
    axes[0, 0].plot(t, res_no_ps["iq"], label="iq")
    axes[0, 0].set_title("No PS: current response")
    axes[0, 0].set_ylabel("Current [A]")
    axes[0, 0].grid(True)
    axes[0, 0].legend()

    axes[0, 1].plot(t, res_ps["id_ref"], label="id_ref")
    axes[0, 1].plot(t, res_ps["id"], label="id")
    axes[0, 1].plot(t, res_ps["iq_ref"], label="iq_ref")
    axes[0, 1].plot(t, res_ps["iq"], label="iq")
    axes[0, 1].set_title("With PS: current response")
    axes[0, 1].set_ylabel("Current [A]")
    axes[0, 1].grid(True)
    axes[0, 1].legend()

    # Row 2: d-axis PS signal
    axes[1, 0].plot(t, res_no_ps["id_ps"], label="id_ps")
    axes[1, 0].set_title("No PS: injected signal")
    axes[1, 0].set_ylabel("id_ps [A]")
    axes[1, 0].grid(True)
    axes[1, 0].legend()

    axes[1, 1].plot(t, res_ps["id_ps"], label="id_ps")
    axes[1, 1].set_title("With PS: injected signal")
    axes[1, 1].set_ylabel("id_ps [A]")
    axes[1, 1].grid(True)
    axes[1, 1].legend()

    # Row 3: parameter estimates
    axes[2, 0].plot(t, res_no_ps["R1_hat"], label="R1_hat")
    axes[2, 0].axhline(p.R1, linestyle="--", label="R1_true")
    axes[2, 0].plot(t, res_no_ps["Ld_hat"] * 1e3, label="Ld_hat [mH]")
    axes[2, 0].axhline(p.Ld * 1e3, linestyle="--", label="Ld_true [mH]")
    axes[2, 0].plot(t, res_no_ps["Lq_hat"] * 1e3, label="Lq_hat [mH]")
    axes[2, 0].axhline(p.Lq * 1e3, linestyle="--", label="Lq_true [mH]")
    axes[2, 0].set_title("No PS: parameter estimates")
    axes[2, 0].set_ylabel("Mixed scale")
    axes[2, 0].grid(True)
    axes[2, 0].legend()

    axes[2, 1].plot(t, res_ps["R1_hat"], label="R1_hat")
    axes[2, 1].axhline(p.R1, linestyle="--", label="R1_true")
    axes[2, 1].plot(t, res_ps["Ld_hat"] * 1e3, label="Ld_hat [mH]")
    axes[2, 1].axhline(p.Ld * 1e3, linestyle="--", label="Ld_true [mH]")
    axes[2, 1].plot(t, res_ps["Lq_hat"] * 1e3, label="Lq_hat [mH]")
    axes[2, 1].axhline(p.Lq * 1e3, linestyle="--", label="Lq_true [mH]")
    axes[2, 1].set_title("With PS: parameter estimates")
    axes[2, 1].set_ylabel("Mixed scale")
    axes[2, 1].grid(True)
    axes[2, 1].legend()

    # Row 4: zeta signals that matter for identifiability
    axes[3, 0].plot(t, res_no_ps["zeta1"], label="zeta1")
    axes[3, 0].plot(t, res_no_ps["zeta2"], label="zeta2")
    axes[3, 0].plot(t, res_no_ps["zeta3"], label="zeta3")
    axes[3, 0].set_title("No PS: regressor signals")
    axes[3, 0].set_ylabel("zeta")
    axes[3, 0].grid(True)
    axes[3, 0].legend()

    axes[3, 1].plot(t, res_ps["zeta1"], label="zeta1")
    axes[3, 1].plot(t, res_ps["zeta2"], label="zeta2")
    axes[3, 1].plot(t, res_ps["zeta3"], label="zeta3")
    axes[3, 1].set_title("With PS: regressor signals")
    axes[3, 1].set_ylabel("zeta")
    axes[3, 1].grid(True)
    axes[3, 1].legend()

    # Row 5: identification error and torque
    axes[4, 0].plot(t, res_no_ps["error"], label="ident error")
    axes[4, 0].plot(t, res_no_ps["torque"], label="torque")
    axes[4, 0].set_title("No PS: error / torque")
    axes[4, 0].set_ylabel("Error / Torque")
    axes[4, 0].set_xlabel("Time [s]")
    axes[4, 0].grid(True)
    axes[4, 0].legend()

    axes[4, 1].plot(t, res_ps["error"], label="ident error")
    axes[4, 1].plot(t, res_ps["torque"], label="torque")
    axes[4, 1].set_title("With PS: error / torque")
    axes[4, 1].set_ylabel("Error / Torque")
    axes[4, 1].set_xlabel("Time [s]")
    axes[4, 1].grid(True)
    axes[4, 1].legend()

    plt.tight_layout()
    plt.show()


def plot_overlay_estimates(res_no_ps: dict, res_ps: dict) -> None:
    t = res_no_ps["t"]
    p = res_no_ps["true_params"]

    fig, axes = plt.subplots(4, 1, figsize=(10, 14), sharex=True)

    axes[0].plot(t, res_no_ps["R1_hat"], label="R1_hat (no PS)")
    axes[0].plot(t, res_ps["R1_hat"], label="R1_hat (with PS)")
    axes[0].axhline(p.R1, linestyle="--", label="R1_true")
    axes[0].set_ylabel("R1 [ohm]")
    axes[0].grid(True)
    axes[0].legend()

    axes[1].plot(t, res_no_ps["Ld_hat"] * 1e3, label="Ld_hat (no PS)")
    axes[1].plot(t, res_ps["Ld_hat"] * 1e3, label="Ld_hat (with PS)")
    axes[1].axhline(p.Ld * 1e3, linestyle="--", label="Ld_true")
    axes[1].set_ylabel("Ld [mH]")
    axes[1].grid(True)
    axes[1].legend()

    axes[2].plot(t, res_no_ps["Lq_hat"] * 1e3, label="Lq_hat (no PS)")
    axes[2].plot(t, res_ps["Lq_hat"] * 1e3, label="Lq_hat (with PS)")
    axes[2].axhline(p.Lq * 1e3, linestyle="--", label="Lq_true")
    axes[2].set_ylabel("Lq [mH]")
    axes[2].grid(True)
    axes[2].legend()

    axes[3].plot(t, res_no_ps["error"], label="error (no PS)")
    axes[3].plot(t, res_ps["error"], label="error (with PS)")
    axes[3].set_ylabel("e_k")
    axes[3].set_xlabel("Time [s]")
    axes[3].grid(True)
    axes[3].legend()

    plt.tight_layout()
    plt.show()


def main() -> None:
    no_ps_cfg = SimCaseConfig(
        name="Case A: No PS",
        enable_ps=False,
    )

    with_ps_cfg = SimCaseConfig(
        name="Case B: With PS",
        enable_ps=True,
    )

    res_no_ps = run_case(no_ps_cfg)
    res_ps = run_case(with_ps_cfg)

    summarize_result(res_no_ps)
    summarize_result(res_ps)

    plot_comparison(res_no_ps, res_ps)
    plot_overlay_estimates(res_no_ps, res_ps)


if __name__ == "__main__":
    main()
```

## この比較版で見るポイント

まず注目するのはこの3つです。

### 1. `zeta1, zeta2, zeta3` の動き

PSなしでは、定常に入ると d軸側の変動が弱くなり、(\zeta) ベクトルの独立性が下がりやすいです。
PSありでは、少なくとも d軸由来の励振が入り、(\zeta_2=sF(i_d)) が明確に動きます。
文献の「定常 dq 系では電流が直流となり、そのままでは複数パラメータ同定の PS 性が不足する」という話を、ここで波形として確認できます。

### 2. `R1_hat, Ld_hat, Lq_hat` の収束の差

PSなしでも、初期過渡や制御誤差で少しは動くことがあります。
ただし、その情報量だけでは 3 パラメータを安定に分離しにくく、PSありの方が収束しやすいはずです。
これは d軸式に基づく 3 パラメータ同定の実験的理解として重要です。

### 3. `error` の減り方

PSありの方が、同定誤差 (e_k) の振る舞いが「意味のある変動を伴ったうえで」落ち着きやすくなります。
PSなしは、見かけ上誤差が小さくても、真にパラメータ分離できていないことがあります。

## ひとつ補足

この最小版は still **かなり理想化**されています。

* 電圧は真値ベースのデカップリングを使っている
* 相殺信号はまだ入れていない
* 高速域での PI 限界もまだ本格的には見ていない
* ノイズ、外乱 (n)、パラメータ突変も未導入

なので、この段階の目的は
**「PSがないと 3 パラメータ同定の回帰ベクトルが貧弱になる」ことを確認する**
に置くのが正しいです。

次は、これに続けて

**相殺信号を入れた版**
または
**パラメータ突変を入れて追従性を見る版**

へ進むのが自然です。
