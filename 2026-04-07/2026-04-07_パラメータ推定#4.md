了解です。
今回は **PS信号を d軸へ入れ、そのトルク脈動を q軸の相殺信号で打ち消す** 比較版にします。

文献の相殺定理では、d軸 PS 確保信号 (i_{dps}^*) に対して、q軸相殺信号 (i_{qps}^*) を

[
i_{qps}^* =
-\frac{2L_m i_{qf}^*}{2L_m(i_{df}^*+i_{dps}^*)+\Phi}, i_{dps}^*
]

で与えると、PS由来のトルクを理論上相殺できます。ここで (L_m=(L_d-L_q)/2) です。
また、PS生成器は d軸基本電流指令に正弦波を重畳して PS 性を作る、という構成です。 

今回は比較しやすいように、同じ `simulation.py` の中で次の3ケースを回します。

* Case A: **PSなし**
* Case B: **PSあり・相殺なし**
* Case C: **PSあり・相殺あり**

`plant.py` と `identifier_d.py` はそのままで使えます。

---

## `simulation.py`

```python
# simulation.py
from __future__ import annotations

import math
from dataclasses import dataclass

import matplotlib.pyplot as plt
import numpy as np

from plant import PMSMParams, PMSMPlantDQ
from identifier_d import IdentifierConfig, ThreeParameterIdentifierD


@dataclass
class SimCaseConfig:
    name: str
    enable_ps: bool
    enable_cancellation: bool

    Ts: float = 1.0e-4
    T_end: float = 1.5
    omega: float = 400.0          # electrical angular speed [rad/s]

    id_base: float = -40.0        # d-axis base current [A]
    iq_base: float = 80.0         # q-axis base current [A]

    omega_ps: float = 50.0        # PS angular frequency [rad/s]
    id_ps_amp: float = 10.0       # PS amplitude [A]

    v_limit: float = 300.0        # voltage saturation [V]

    # minimal PI controller
    kp_d: float = 8.0
    ki_d: float = 400.0
    kp_q: float = 8.0
    ki_q: float = 400.0


def cancellation_signal(
    idf: float,
    iqf: float,
    idps: float,
    Ld: float,
    Lq: float,
    Phi: float,
    eps: float = 1e-9,
) -> float:
    """
    Cancellation signal based on the paper's cancellation theorem.

    iqps = -(2*Lm*iqf)/(2*Lm*(idf + idps) + Phi) * idps
    where Lm = (Ld - Lq)/2
    """
    Lm = 0.5 * (Ld - Lq)
    den = 2.0 * Lm * (idf + idps) + Phi

    if abs(den) < eps:
        den = eps if den >= 0.0 else -eps

    iqps = -(2.0 * Lm * iqf / den) * idps
    return iqps


def run_case(cfg: SimCaseConfig) -> dict:
    Ts = cfg.Ts
    steps = int(cfg.T_end / Ts)
    t = np.arange(steps) * Ts

    # True motor parameters
    true_params = PMSMParams(
        R1=0.03,
        Ld=1.00e-3,
        Lq=1.50e-3,
        Phi=0.10,
    )

    plant = PMSMPlantDQ(true_params, Ts)

    ident = ThreeParameterIdentifierD(
        IdentifierConfig(
            Ts=Ts,
            omega_ss3=2.0 * cfg.omega_ps,
            lambda1=0.99,
            lambda2=1.0,
            gsig=100.0,
            nominal_R1=0.03,
            nominal_Ld=1.00e-3,
            nominal_Lq=1.50e-3,
            use_output_lpf=False,
            omega_out=100.0,
        )
    )

    int_d = 0.0
    int_q = 0.0

    # Logs
    id_log = np.zeros(steps)
    iq_log = np.zeros(steps)

    id_ref_log = np.zeros(steps)
    iq_ref_log = np.zeros(steps)

    id_ps_log = np.zeros(steps)
    iq_ps_log = np.zeros(steps)

    vd_log = np.zeros(steps)
    vq_log = np.zeros(steps)

    R1_hat_log = np.zeros(steps)
    Ld_hat_log = np.zeros(steps)
    Lq_hat_log = np.zeros(steps)
    err_log = np.zeros(steps)

    zeta0_log = np.zeros(steps)
    zeta1_log = np.zeros(steps)
    zeta2_log = np.zeros(steps)
    zeta3_log = np.zeros(steps)

    torque_log = np.zeros(steps)
    torque_ripple_log = np.zeros(steps)

    for k in range(steps):
        tk = t[k]

        # Base commands
        idf = cfg.id_base
        iqf = cfg.iq_base

        # d-axis PS signal
        if cfg.enable_ps:
            idps = cfg.id_ps_amp * math.sin(cfg.omega_ps * tk)
        else:
            idps = 0.0

        # q-axis cancellation signal
        if cfg.enable_ps and cfg.enable_cancellation:
            # Minimal understanding version:
            # use true parameters for cancellation synthesis first.
            # Later this can be replaced with identified values.
            iqps = cancellation_signal(
                idf=idf,
                iqf=iqf,
                idps=idps,
                Ld=true_params.Ld,
                Lq=true_params.Lq,
                Phi=true_params.Phi,
            )
        else:
            iqps = 0.0

        id_ref = idf + idps
        iq_ref = iqf + iqps

        # Current controller
        e_d = id_ref - plant.id
        e_q = iq_ref - plant.iq

        int_d += Ts * e_d
        int_q += Ts * e_q

        # Feedforward / decoupling using true model
        vd_ff = true_params.R1 * plant.id - cfg.omega * true_params.Lq * plant.iq
        vq_ff = (
            true_params.R1 * plant.iq
            + cfg.omega * true_params.Ld * plant.id
            + cfg.omega * true_params.Phi
        )

        vd = vd_ff + cfg.kp_d * e_d + cfg.ki_d * int_d
        vq = vq_ff + cfg.kp_q * e_q + cfg.ki_q * int_q

        vd = float(np.clip(vd, -cfg.v_limit, cfg.v_limit))
        vq = float(np.clip(vq, -cfg.v_limit, cfg.v_limit))

        id_now, iq_now = plant.step(vd, vq, cfg.omega)
        torque = plant.electromagnetic_torque(pole_pairs=1)

        est = ident.update(vd=vd, id_=id_now, iq=iq_now, omega=cfg.omega)

        # torque ripple relative to base-torque reference
        # base torque = Np * (Phi + (Ld-Lq)*idf) * iqf
        torque_base = (true_params.Phi + (true_params.Ld - true_params.Lq) * idf) * iqf
        torque_ripple = torque - torque_base

        # Logs
        id_log[k] = id_now
        iq_log[k] = iq_now

        id_ref_log[k] = id_ref
        iq_ref_log[k] = iq_ref

        id_ps_log[k] = idps
        iq_ps_log[k] = iqps

        vd_log[k] = vd
        vq_log[k] = vq

        R1_hat_log[k] = est["R1_hat"]
        Ld_hat_log[k] = est["Ld_hat"]
        Lq_hat_log[k] = est["Lq_hat"]
        err_log[k] = est["error"]

        zeta0_log[k] = est["zeta0"]
        zeta1_log[k] = est["zeta1"]
        zeta2_log[k] = est["zeta2"]
        zeta3_log[k] = est["zeta3"]

        torque_log[k] = torque
        torque_ripple_log[k] = torque_ripple

    return {
        "cfg": cfg,
        "t": t,
        "true_params": true_params,
        "id": id_log,
        "iq": iq_log,
        "id_ref": id_ref_log,
        "iq_ref": iq_ref_log,
        "id_ps": id_ps_log,
        "iq_ps": iq_ps_log,
        "vd": vd_log,
        "vq": vq_log,
        "R1_hat": R1_hat_log,
        "Ld_hat": Ld_hat_log,
        "Lq_hat": Lq_hat_log,
        "error": err_log,
        "zeta0": zeta0_log,
        "zeta1": zeta1_log,
        "zeta2": zeta2_log,
        "zeta3": zeta3_log,
        "torque": torque_log,
        "torque_ripple": torque_ripple_log,
    }


def summarize_result(result: dict) -> None:
    p = result["true_params"]
    ripple_rms = float(np.sqrt(np.mean(result["torque_ripple"] ** 2)))
    print(f"=== {result['cfg'].name} ===")
    print(f"R1 true={p.R1:.6f}, hat={result['R1_hat'][-1]:.6f}")
    print(f"Ld true={p.Ld:.6e}, hat={result['Ld_hat'][-1]:.6e}")
    print(f"Lq true={p.Lq:.6e}, hat={result['Lq_hat'][-1]:.6e}")
    print(f"final ident error={result['error'][-1]:.6e}")
    print(f"torque ripple RMS={ripple_rms:.6e}")
    print()


def plot_three_case_comparison(res_a: dict, res_b: dict, res_c: dict) -> None:
    t = res_a["t"]
    p = res_a["true_params"]

    fig, axes = plt.subplots(5, 3, figsize=(18, 18), sharex="col")

    cases = [res_a, res_b, res_c]
    titles = [
        "Case A: No PS",
        "Case B: With PS / No cancellation",
        "Case C: With PS / With cancellation",
    ]

    for col, (res, title) in enumerate(zip(cases, titles)):
        # Row 1: current references and responses
        axes[0, col].plot(t, res["id_ref"], label="id_ref")
        axes[0, col].plot(t, res["id"], label="id")
        axes[0, col].plot(t, res["iq_ref"], label="iq_ref")
        axes[0, col].plot(t, res["iq"], label="iq")
        axes[0, col].set_title(title)
        axes[0, col].set_ylabel("Current [A]")
        axes[0, col].grid(True)
        axes[0, col].legend()

        # Row 2: injected signals
        axes[1, col].plot(t, res["id_ps"], label="id_ps")
        axes[1, col].plot(t, res["iq_ps"], label="iq_ps")
        axes[1, col].set_ylabel("Injected [A]")
        axes[1, col].grid(True)
        axes[1, col].legend()

        # Row 3: parameter estimates
        axes[2, col].plot(t, res["R1_hat"], label="R1_hat")
        axes[2, col].axhline(p.R1, linestyle="--", label="R1_true")
        axes[2, col].plot(t, res["Ld_hat"] * 1e3, label="Ld_hat [mH]")
        axes[2, col].axhline(p.Ld * 1e3, linestyle="--", label="Ld_true [mH]")
        axes[2, col].plot(t, res["Lq_hat"] * 1e3, label="Lq_hat [mH]")
        axes[2, col].axhline(p.Lq * 1e3, linestyle="--", label="Lq_true [mH]")
        axes[2, col].set_ylabel("Mixed scale")
        axes[2, col].grid(True)
        axes[2, col].legend()

        # Row 4: regressor signals
        axes[3, col].plot(t, res["zeta1"], label="zeta1")
        axes[3, col].plot(t, res["zeta2"], label="zeta2")
        axes[3, col].plot(t, res["zeta3"], label="zeta3")
        axes[3, col].set_ylabel("zeta")
        axes[3, col].grid(True)
        axes[3, col].legend()

        # Row 5: torque ripple and ident error
        axes[4, col].plot(t, res["torque_ripple"], label="torque_ripple")
        axes[4, col].plot(t, res["error"], label="ident_error")
        axes[4, col].set_ylabel("Ripple / Error")
        axes[4, col].set_xlabel("Time [s]")
        axes[4, col].grid(True)
        axes[4, col].legend()

    plt.tight_layout()
    plt.show()


def plot_torque_overlay(res_a: dict, res_b: dict, res_c: dict) -> None:
    t = res_a["t"]

    plt.figure(figsize=(10, 6))
    plt.plot(t, res_a["torque_ripple"], label="No PS")
    plt.plot(t, res_b["torque_ripple"], label="PS / no cancellation")
    plt.plot(t, res_c["torque_ripple"], label="PS / with cancellation")
    plt.ylabel("Torque ripple")
    plt.xlabel("Time [s]")
    plt.grid(True)
    plt.legend()
    plt.tight_layout()
    plt.show()


def main() -> None:
    case_a = SimCaseConfig(
        name="Case A: No PS",
        enable_ps=False,
        enable_cancellation=False,
    )

    case_b = SimCaseConfig(
        name="Case B: With PS / No cancellation",
        enable_ps=True,
        enable_cancellation=False,
    )

    case_c = SimCaseConfig(
        name="Case C: With PS / With cancellation",
        enable_ps=True,
        enable_cancellation=True,
    )

    res_a = run_case(case_a)
    res_b = run_case(case_b)
    res_c = run_case(case_c)

    summarize_result(res_a)
    summarize_result(res_b)
    summarize_result(res_c)

    plot_three_case_comparison(res_a, res_b, res_c)
    plot_torque_overlay(res_a, res_b, res_c)


if __name__ == "__main__":
    main()
```

---

## この版で見るポイント

まず最重要なのは **`torque_ripple`** です。

* **Case A: No PS**
  基本的に PS がないので、PS起因のトルク脈動は出ません。
* **Case B: With PS / No cancellation**
  d軸へ PS を入れたぶん、突極性由来のトルク脈動が出ます。
* **Case C: With PS / With cancellation**
  q軸相殺信号を入れることで、その脈動が下がるはずです。

つまりこの版の狙いは、
**「PSで可同定性を作るとトルクが揺れるが、相殺信号で抑えられる」**
をまず数値で見ることです。これは文献の 3 と 4 の項目そのものです。 

---

## 今回の実装上の割り切り

この版では、相殺信号の計算に **真値パラメータ** を使っています。
理由は、最初の理解ステップとして

1. 相殺式そのものが効くか
2. トルク脈動が本当に減るか

をきれいに見るためです。

文献の最終構成では、ここは当然 **同定値** に置き換えて使う想定です。
次の段階では、`Ld, Lq` に `L̂d, L̂q` を使い、さらに `Phi` は固定値または後段の (\hat{\Phi}) に置き換える、という流れになります。

---

## 次に自然な拡張

次の一手として一番よいのは、相殺信号の計算を

* `Ld, Lq` は **同定値**
* `Phi` は **真値固定** か **仮値固定**

に置き換えた版にすることです。

そうすると、
**「同定値を使った PS生成器」**
に一歩近づきます。
