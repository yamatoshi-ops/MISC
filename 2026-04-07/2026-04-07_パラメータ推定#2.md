まずは最小構成でいきましょう。
以下の3ファイルで、**d軸回路方程式に基づく 3 パラメータ同定**だけを単体で回せます。

この最小構成は、文献の

* d軸式
  (v_d = R_1 i_d + L_d \frac{di_d}{dt} - \omega L_q i_q + n)
* 整形分離フィルタで
  (\zeta_0 = R_1 \zeta_1 + L_d \zeta_2 + L_q \zeta_3)
* 最小二乗形一般化適応同定アルゴリズム

という流れを、まず Python 上で確認するためのものです。  

今回はあえて簡略化しています。

* プラントは dq の理想モデル
* 電圧は「簡易 PI + 既知構造の補償」で生成
* PS 性確保のため d軸に正弦波電流を重畳
* 相殺信号、q軸の (\Phi) 同定、2次制御器はまだ入れない

## `plant.py`

```python
# plant.py
from __future__ import annotations

from dataclasses import dataclass


@dataclass
class PMSMParams:
    R1: float     # [ohm]
    Ld: float     # [H]
    Lq: float     # [H]
    Phi: float    # [Wb]


class PMSMPlantDQ:
    """
    Minimal PMSM dq plant.

    Continuous-time model:
        vd = R1*id + Ld*did/dt - omega*Lq*iq
        vq = R1*iq + Lq*diq/dt + omega*Ld*id + omega*Phi

    Rearranged:
        did/dt = (vd - R1*id + omega*Lq*iq) / Ld
        diq/dt = (vq - R1*iq - omega*Ld*id - omega*Phi) / Lq

    Notes:
    - Uses static inductances.
    - Euler integration.
    - omega is electrical angular speed [rad/s].
    """

    def __init__(self, params: PMSMParams, Ts: float) -> None:
        self.params = params
        self.Ts = Ts

        self.id = 0.0
        self.iq = 0.0

    def reset(self, id0: float = 0.0, iq0: float = 0.0) -> None:
        self.id = id0
        self.iq = iq0

    def step(self, vd: float, vq: float, omega: float) -> tuple[float, float]:
        p = self.params

        did_dt = (vd - p.R1 * self.id + omega * p.Lq * self.iq) / p.Ld
        diq_dt = (vq - p.R1 * self.iq - omega * p.Ld * self.id - omega * p.Phi) / p.Lq

        self.id += self.Ts * did_dt
        self.iq += self.Ts * diq_dt

        return self.id, self.iq

    def electromagnetic_torque(self, pole_pairs: int = 1) -> float:
        """
        tau = Np * (Phi + (Ld - Lq)*id) * iq
        = Np * (Phi - 2*Lm*id) * iq, with Lm=(Ld-Lq)/2
        """
        p = self.params
        return pole_pairs * (p.Phi + (p.Ld - p.Lq) * self.id) * self.iq
```

## `identifier_d.py`

```python
# identifier_d.py
from __future__ import annotations

from dataclasses import dataclass
import numpy as np


@dataclass
class IdentifierConfig:
    Ts: float
    omega_ss3: float = 100.0   # shape-separating filter bandwidth [rad/s]
    lambda1: float = 0.99
    lambda2: float = 1.0
    gsig: float = 100.0        # scalar gain for resistance channel

    nominal_R1: float = 0.03
    nominal_Ld: float = 1.0e-3
    nominal_Lq: float = 1.5e-3

    use_output_lpf: bool = False
    omega_out: float = 100.0   # optional output LPF bandwidth


class FirstOrderLPF:
    """
    y_dot = omega * (u - y)
    Backward-Euler style discrete update.
    """

    def __init__(self, Ts: float, omega: float, x0: float = 0.0) -> None:
        self.Ts = Ts
        self.omega = omega
        self.x = x0

    def reset(self, x0: float = 0.0) -> None:
        self.x = x0

    def update(self, u: float) -> float:
        a = self.omega * self.Ts / (1.0 + self.omega * self.Ts)
        self.x = self.x + a * (u - self.x)
        return self.x


class ThreeParameterIdentifierD:
    """
    d-axis 3-parameter adaptive identifier.

    Target parameters:
        theta = [R1, Ld, Lq]^T

    Identification model:
        zeta0 = zeta^T theta

    with
        zeta0 = F(vd)
        zeta1 = gsig * F(id)
        zeta2 = sF(id)
        zeta3 = F(-omega * iq)

    Notes:
    - gsig is applied only to the resistance channel signal generation.
    - Therefore the internal estimated first parameter corresponds to (R1 / gsig).
      Final R1 estimate is recovered by:
          R1_hat = gsig * theta_hat_internal[0]
    """

    def __init__(self, cfg: IdentifierConfig) -> None:
        self.cfg = cfg

        self.f_vd = FirstOrderLPF(cfg.Ts, cfg.omega_ss3)
        self.f_id = FirstOrderLPF(cfg.Ts, cfg.omega_ss3)
        self.f_wiq = FirstOrderLPF(cfg.Ts, cfg.omega_ss3)

        self.out_R = FirstOrderLPF(cfg.Ts, cfg.omega_out, cfg.nominal_R1)
        self.out_Ld = FirstOrderLPF(cfg.Ts, cfg.omega_out, cfg.nominal_Ld)
        self.out_Lq = FirstOrderLPF(cfg.Ts, cfg.omega_out, cfg.nominal_Lq)

        self.theta_hat = np.zeros((3, 1), dtype=float)
        self.Gamma = np.eye(3, dtype=float)

        self.prev_f_id = 0.0

        self.limit_R1 = (0.1 * cfg.nominal_R1, 5.0 * cfg.nominal_R1)
        self.limit_Ld = (0.1 * cfg.nominal_Ld, 2.0 * cfg.nominal_Ld)
        self.limit_Lq = (0.1 * cfg.nominal_Lq, 2.0 * cfg.nominal_Lq)

    def reset(self) -> None:
        self.f_vd.reset(0.0)
        self.f_id.reset(0.0)
        self.f_wiq.reset(0.0)

        self.out_R.reset(self.cfg.nominal_R1)
        self.out_Ld.reset(self.cfg.nominal_Ld)
        self.out_Lq.reset(self.cfg.nominal_Lq)

        self.theta_hat[:] = 0.0
        self.Gamma[:] = np.eye(3, dtype=float)
        self.prev_f_id = 0.0

    def update(self, vd: float, id_: float, iq: float, omega: float) -> dict:
        # Shape-separating filter outputs
        zeta0 = self.f_vd.update(vd)

        f_id_now = self.f_id.update(id_)
        zeta1 = self.cfg.gsig * f_id_now

        # sF(id) ≈ discrete derivative of filtered id
        zeta2 = (f_id_now - self.prev_f_id) / self.cfg.Ts
        self.prev_f_id = f_id_now

        zeta3 = self.f_wiq.update(-omega * iq)

        zeta = np.array([[zeta1], [zeta2], [zeta3]], dtype=float)
        zeta0_vec = np.array([[zeta0]], dtype=float)

        # Least-squares-type generalized adaptive identification algorithm
        e = float(zeta.T @ self.theta_hat - zeta0_vec)

        den = self.cfg.lambda1 + self.cfg.lambda2 * float(zeta.T @ self.Gamma @ zeta)
        gain_vec = (self.cfg.lambda2 * self.Gamma @ zeta) / den

        self.theta_hat = self.theta_hat - gain_vec * e
        self.Gamma = (1.0 / self.cfg.lambda1) * (
            self.Gamma
            - (self.cfg.lambda2 * self.Gamma @ zeta @ zeta.T @ self.Gamma) / den
        )

        # Recover physical parameters
        R1_hat_raw = self.cfg.gsig * float(self.theta_hat[0, 0])
        Ld_hat_raw = float(self.theta_hat[1, 0])
        Lq_hat_raw = float(self.theta_hat[2, 0])

        # Limiter
        R1_hat_lim = float(np.clip(R1_hat_raw, *self.limit_R1))
        Ld_hat_lim = float(np.clip(Ld_hat_raw, *self.limit_Ld))
        Lq_hat_lim = float(np.clip(Lq_hat_raw, *self.limit_Lq))

        # Optional output LPF
        if self.cfg.use_output_lpf:
            R1_hat = self.out_R.update(R1_hat_lim)
            Ld_hat = self.out_Ld.update(Ld_hat_lim)
            Lq_hat = self.out_Lq.update(Lq_hat_lim)
        else:
            R1_hat = R1_hat_lim
            Ld_hat = Ld_hat_lim
            Lq_hat = Lq_hat_lim

        return {
            "R1_hat": R1_hat,
            "Ld_hat": Ld_hat,
            "Lq_hat": Lq_hat,
            "zeta0": zeta0,
            "zeta1": zeta1,
            "zeta2": zeta2,
            "zeta3": zeta3,
            "error": e,
            "theta_hat_internal": self.theta_hat.copy(),
            "Gamma_diag": np.diag(self.Gamma).copy(),
        }
```

## `simulation.py`

```python
# simulation.py
from __future__ import annotations

import math
import numpy as np
import matplotlib.pyplot as plt

from plant import PMSMParams, PMSMPlantDQ
from identifier_d import IdentifierConfig, ThreeParameterIdentifierD


def run_simulation() -> None:
    # ------------------------------------------------------------------
    # 1) Simulation settings
    # ------------------------------------------------------------------
    Ts = 1.0e-4           # [s]
    T_end = 1.5           # [s]
    steps = int(T_end / Ts)

    omega = 400.0         # electrical speed [rad/s]
    t = np.arange(steps) * Ts

    # True motor parameters
    true_params = PMSMParams(
        R1=0.03,
        Ld=1.00e-3,
        Lq=1.50e-3,
        Phi=0.10,
    )

    plant = PMSMPlantDQ(true_params, Ts)

    # ------------------------------------------------------------------
    # 2) Identifier settings
    #    Paper-side example values often use:
    #      omega_ps = 50 rad/s
    #      omega_ss3 = 2*omega_ps = 100 rad/s
    #      lambda1 = 0.99, lambda2 = 1
    #      gsig = 100
    # ------------------------------------------------------------------
    omega_ps = 50.0
    id_ps_amp = 10.0      # [A]

    ident = ThreeParameterIdentifierD(
        IdentifierConfig(
            Ts=Ts,
            omega_ss3=2.0 * omega_ps,
            lambda1=0.99,
            lambda2=1.0,
            gsig=100.0,
            nominal_R1=0.03,
            nominal_Ld=1.00e-3,
            nominal_Lq=1.50e-3,
            use_output_lpf=False,
            omega_out=100.0,
        )
    )

    # ------------------------------------------------------------------
    # 3) Simple current control for excitation generation
    #
    #    This is intentionally minimal:
    #    - track id* and iq* with PI
    #    - add decoupling/feedforward using true parameters
    #
    #    For a first-step identification demo, this is acceptable.
    # ------------------------------------------------------------------
    id_base = -40.0
    iq_ref = 80.0

    kp_d = 8.0
    ki_d = 400.0
    kp_q = 8.0
    ki_q = 400.0

    int_d = 0.0
    int_q = 0.0

    # voltage saturation
    v_limit = 300.0

    # ------------------------------------------------------------------
    # 4) Logging
    # ------------------------------------------------------------------
    id_log = np.zeros(steps)
    iq_log = np.zeros(steps)
    id_ref_log = np.zeros(steps)
    iq_ref_log = np.zeros(steps)
    vd_log = np.zeros(steps)
    vq_log = np.zeros(steps)

    R1_hat_log = np.zeros(steps)
    Ld_hat_log = np.zeros(steps)
    Lq_hat_log = np.zeros(steps)
    err_log = np.zeros(steps)

    zeta0_log = np.zeros(steps)
    zeta1_log = np.zeros(steps)
    zeta2_log = np.zeros(steps)
    zeta3_log = np.zeros(steps)

    torque_log = np.zeros(steps)

    # ------------------------------------------------------------------
    # 5) Main simulation
    # ------------------------------------------------------------------
    for k in range(steps):
        tk = t[k]

        # Persistent excitation on d-axis
        id_ps = id_ps_amp * math.sin(omega_ps * tk)
        id_ref = id_base + id_ps

        # Keep iq simple and constant in this minimal example
        iq_ref = iq_ref

        # Current control
        e_d = id_ref - plant.id
        e_q = iq_ref - plant.iq

        int_d += Ts * e_d
        int_q += Ts * e_q

        # Decoupling / feedforward using true model
        vd_ff = true_params.R1 * plant.id - omega * true_params.Lq * plant.iq
        vq_ff = true_params.R1 * plant.iq + omega * true_params.Ld * plant.id + omega * true_params.Phi

        vd = vd_ff + kp_d * e_d + ki_d * int_d
        vq = vq_ff + kp_q * e_q + ki_q * int_q

        vd = float(np.clip(vd, -v_limit, v_limit))
        vq = float(np.clip(vq, -v_limit, v_limit))

        id_now, iq_now = plant.step(vd, vq, omega)
        torque = plant.electromagnetic_torque(pole_pairs=1)

        est = ident.update(vd=vd, id_=id_now, iq=iq_now, omega=omega)

        # Log
        id_log[k] = id_now
        iq_log[k] = iq_now
        id_ref_log[k] = id_ref
        iq_ref_log[k] = iq_ref
        vd_log[k] = vd
        vq_log[k] = vq

        R1_hat_log[k] = est["R1_hat"]
        Ld_hat_log[k] = est["Ld_hat"]
        Lq_hat_log[k] = est["Lq_hat"]
        err_log[k] = est["error"]

        zeta0_log[k] = est["zeta0"]
        zeta1_log[k] = est["zeta1"]
        zeta2_log[k] = est["zeta2"]
        zeta3_log[k] = est["zeta3"]

        torque_log[k] = torque

    # ------------------------------------------------------------------
    # 6) Print final estimates
    # ------------------------------------------------------------------
    print("=== Final estimates ===")
    print(f"R1 true={true_params.R1:.6f}, hat={R1_hat_log[-1]:.6f}")
    print(f"Ld true={true_params.Ld:.6e}, hat={Ld_hat_log[-1]:.6e}")
    print(f"Lq true={true_params.Lq:.6e}, hat={Lq_hat_log[-1]:.6e}")

    # ------------------------------------------------------------------
    # 7) Plot
    # ------------------------------------------------------------------
    fig, axes = plt.subplots(4, 1, figsize=(10, 14), sharex=True)

    axes[0].plot(t, id_ref_log, label="id_ref")
    axes[0].plot(t, id_log, label="id")
    axes[0].plot(t, iq_ref_log, label="iq_ref")
    axes[0].plot(t, iq_log, label="iq")
    axes[0].set_ylabel("Current [A]")
    axes[0].grid(True)
    axes[0].legend()

    axes[1].plot(t, R1_hat_log, label="R1_hat")
    axes[1].axhline(true_params.R1, linestyle="--", label="R1_true")
    axes[1].set_ylabel("R1 [ohm]")
    axes[1].grid(True)
    axes[1].legend()

    axes[2].plot(t, Ld_hat_log * 1e3, label="Ld_hat")
    axes[2].plot(t, Lq_hat_log * 1e3, label="Lq_hat")
    axes[2].axhline(true_params.Ld * 1e3, linestyle="--", label="Ld_true")
    axes[2].axhline(true_params.Lq * 1e3, linestyle="--", label="Lq_true")
    axes[2].set_ylabel("Inductance [mH]")
    axes[2].grid(True)
    axes[2].legend()

    axes[3].plot(t, err_log, label="identification error e_k")
    axes[3].plot(t, torque_log, label="torque")
    axes[3].set_ylabel("Error / Torque")
    axes[3].set_xlabel("Time [s]")
    axes[3].grid(True)
    axes[3].legend()

    plt.tight_layout()
    plt.show()


if __name__ == "__main__":
    run_simulation()
```

## この実装の割り切り

この版は、あくまで**文献理解のための最小版**です。
まだ次は入れていません。

* q軸の (\Phi) 同定
* 相殺信号
* 2次電流制御器
* 出力処理の 2次 LPF
* 真の電圧検出値の使用
* パラメータ突変試験
